save_plots <- function(
    filename,      # 基础文件名(不含扩展名)
    plot,          # 要保存的图形对象(ggplot或grid对象)
    path = ".",    # 保存路径,默认当前目录
    width = 7.2,   # 宽度(英寸)
    height = 4.8,  # 高度(英寸)
    dpi = 600,     # PNG分辨率
    formats = c("pdf", "png")  # 支持保存的格式
) {
  library(ggplot2)
  library(grid)
  
  # 创建目录(如果不存在)
  if (!dir.exists(path)) {
    dir.create(path, recursive = TRUE)
  }
  
  # 判断是否为ggplot对象
  is_ggplot <- inherits(plot, "ggplot")
  
  # 遍历所有需要保存的格式
  for (fmt in formats) {
    full_path <- file.path(path, paste0(filename, ".", fmt))
    
    if (is_ggplot) {
      # 如果是ggplot对象,使用ggsave
      switch(fmt,
             "pdf" = ggsave(full_path, plot, width = width, height = height),
             "png" = ggsave(full_path, plot, width = width, height = height, dpi = dpi, bg = "white"),
             "tiff" = ggsave(full_path, plot, width = width, height = height, dpi = dpi, bg = "white"),
             warning(paste("Unsupported format:", fmt))
      )
    } else {
      # 如果是grid对象(如pheatmap),使用传统图形设备
      switch(fmt,
             "pdf" = {
               pdf(full_path, width = width, height = height)
               print(plot)
               dev.off()
             },
             "png" = {
               png(full_path, width = width * dpi, height = height * dpi, res = dpi, bg = "white")
               print(plot)
               dev.off()
             },
             "tiff" = {
               tiff(full_path, width = width * dpi, height = height * dpi, res = dpi, bg = "white")
               print(plot)
               dev.off()
             },
             warning(paste("Unsupported format:", fmt))
      )
    }
    
    message(paste("Saved:", full_path))
  }
  
  invisible(NULL)
}

message(paste0("实验组 (ident.1): ", group_exp))

# 要分析的细胞类型列表
cell_types_to_analyze <- unique(merged$Celltype)

# 初始化序号计数器（从00开始）
counter <- 0

# 遍历每个细胞类型
for (cell_type in cell_types_to_analyze) {
  
  message(paste0("\n========== 分析细胞类型: ", cell_type, " =========="))
  
  # 生成序号前缀（00, 01, 02...）
  id_str <- sprintf("%02d", counter)
  
  # 清理细胞类型名称用于文件名（替换特殊字符为下划线）
  cleaned_ct <- gsub("[^a-zA-Z0-9]", "_", cell_type)
  
  # 1. 提取特定细胞类型子集
  sub_obj <- subset(merged, subset = Celltype == cell_type)
  
  # 检查分组情况
  cell_counts <- table(sub_obj$Group)
  message("细胞数量: ", paste(names(cell_counts), cell_counts, sep = ":", collapse = ", "))
  
  if (length(cell_counts) < 2 || any(cell_counts < 3)) {
    warning(paste0(cell_type, ": 某组细胞数量不足，跳过"))
    counter <- counter + 1
    next
  }
  
  # 2. 多组差异分析（每组 vs 其他所有组）
  message("运行多组差异分析 (FindAllMarkers)...")
  de_genes_all <- FindAllMarkers(
    sub_obj,
    group.by = "Group",     # 明确指定分组变量
    min.pct = 0.4,
    logfc.threshold = 0.5,
    only.pos = FALSE,       # 同时获取上调和下调基因（后续会分开）
    test.use = "wilcox"     # 可根据需要调整
  )
  
  # 查看结果结构
  # head(de_genes_all)
  # 列包括: gene, avg_log2FC, pct.1, pct.2, p_val_adj, group (该基因在哪个组中显著)
  
  if (nrow(de_genes_all) == 0) {
    warning(paste0(cell_type, ": 未找到任何差异基因"))
    counter <- counter + 1
    next
  }
  
  message(paste0("总差异基因数: ", nrow(de_genes_all)))
  
  colnames(de_genes_all)[colnames(de_genes_all) == "cluster"] <- "group"
  
  # 3. 为每组分别准备基因列表
  DEG_list <- list()
  
  for (grp in unique(de_genes_all$group)) {
    grp_genes <- de_genes_all %>%
      filter(group == grp) %>%
      arrange(desc(avg_log2FC)) %>%
      pull(gene)
    
    if (length(grp_genes) > 0) {
      DEG_list[[grp]] <- grp_genes
      message(paste0(grp, " 高表达基因数: ", length(grp_genes)))
    }
  }
  
  if (length(DEG_list) == 0) {
    warning(paste0(cell_type, ": 无有效基因列表"))
    counter <- counter + 1
    next
  }
  
  # 4. ID转换
  message("转换ENTREZ ID...")
  DEG_list_entrez <- list()
  
  for (grp in names(DEG_list)) {
    genes_symbol <- DEG_list[[grp]]
    
    # 测试 bitr 是否能正常工作
    test_genes <- head(genes_symbol, 100)  # 测试前100个
    test_bitr <- tryCatch(
      bitr(test_genes, fromType = "SYMBOL", toType = "ENTREZID", OrgDb = org_db),
      error = function(e) {
        message(paste0("组 ", grp, " ID转换测试失败: ", e$message))
        return(NULL)
      }
    )
    
    if (is.null(test_bitr) || nrow(test_bitr) == 0) {
      message(paste0(grp, " ID转换测试失败，跳过"))
      next
    }
    
    # 正式转换所有基因
    entrez_df <- tryCatch(
      bitr(genes_symbol, fromType = "SYMBOL", toType = "ENTREZID", OrgDb = org_db),

        message(paste0("组 ", grp, " ID转换失败: ", e$message))
        return(NULL)
      }

    if (is.null(entrez_df) || nrow(entrez_df) == 0) {

    }
    
    entrez_ids <- entrez_df$ENTREZID
    entrez_ids <- unique(entrez_ids[!is.na(entrez_ids)])
    
    if (length(entrez_ids) >= 3) {  # 降低阈值
      DEG_list_entrez[[grp]] <- entrez_ids
      message(paste0(grp, " 成功映射: ", length(entrez_ids), " 个基因"))
    } else {
      message(paste0(grp, " 映射基因数过少 (", length(entrez_ids), ")，跳过富集"))
    }
  }
  
  if (length(DEG_list_entrez) == 0) {
    warning(paste0(cell_type, ": 无足够基因进行富集分析"))
    counter <- counter + 1
    next
  }
  
  # ========== 5. KEGG富集（每组单独） ==========
  if (length(DEG_list_entrez) >= 1) {
    comp_kegg <- compareCluster(
      geneClusters = DEG_list_entrez,
      fun = "enrichKEGG",
      organism = kegg_organism,
      pAdjustMethod = "BH",
      pvalueCutoff = 0.05,
      qvalueCutoff = 0.2
    )
    
    if (!is.null(comp_kegg) && nrow(as.data.frame(comp_kegg)) > 0) {
      comp_kegg <- setReadable(comp_kegg, OrgDb = org_db, keyType = "ENTREZID")
      kegg_df_full <- as.data.frame(comp_kegg)
      
      # 去除KEGG通路名称中的物种后缀
      suffix_to_remove <- if (species == "Human") {
        " - Homo sapiens \\(human\\)"
      } else if (species == "Mouse") {
        " - Mus musculus \\(house mouse\\)"
      } else if (species == "Rat") {
        " - Rattus norvegicus \\(rat\\)"
      } else {
        ""  # 其他物种可能没有标准后缀
      }
      kegg_df_full$Description <- sub(suffix_to_remove, "", kegg_df_full$Description)
      
      # 筛选每组前10条

library(CellChat)
run_cellchat <- function(seurat_obj, group_name, celltype_col = "celltype", organism = "human") {
  
  cat(paste0(">>> Start processing: ", group_name, " <<<\n"))
  
  # 1. 处理rat物种：将基因名映射为小鼠基因
  if (organism == "rat") {
    cat(">>> Converting rat genes to mouse genes <<<\n")
    
    # 基因转换：rat -> mouse
    res <- GeneConvert(
      geneID = rownames(seurat_obj),
      geneID_from_IDtype = "symbol",
      geneID_to_IDtype = "symbol",
      species_from = "Rattus_norvegicus",
      species_to = "Mus_musculus"
    )
    
    # 使用collapse版本
    gene_mapping <- res$geneID_collapse
    
    # 检查列名
    cat(">>> Column names in gene_mapping: <<<\n")
    print(colnames(gene_mapping))
    print(head(gene_mapping))
    
    # 正确的列名应该是 'symbol' (目标基因名)
    # rownames 是原始基因名（rat）
    # symbol 列是转换后的基因名（mouse）
    
    # 过滤掉NA或空值
    valid_mapping <- gene_mapping[!is.na(gene_mapping$symbol) & 
                                    gene_mapping$symbol != "" &
                                    nchar(as.character(gene_mapping$symbol)) > 0, ]
    
    cat(paste0(">>> Valid mappings: ", nrow(valid_mapping), " <<<\n"))
    
    if (nrow(valid_mapping) == 0) {
      stop("No valid gene mappings found!")
    }
    
    # 提取旧基因名（rat）和新基因名（mouse）
    old_names <- rownames(valid_mapping)  # rat基因名
    new_names_vec <- as.character(valid_mapping$symbol)  # mouse基因名
    
    cat(paste0(">>> old_names length: ", length(old_names), " <<<\n"))
    cat(paste0(">>> new_names_vec length: ", length(new_names_vec), " <<<\n"))
    cat(">>> First few mappings: <<<\n")
    print(head(data.frame(rat = old_names, mouse = new_names_vec)))
    
    # 创建命名向量：新名字 = 旧名字
    rename_vector <- setNames(new_names_vec, old_names)
    
    # 过滤Seurat对象中存在的基因
    genes_to_keep <- intersect(rownames(seurat_obj), names(rename_vector))
    
    if (length(genes_to_keep) == 0) {
      stop("No genes could be mapped from rat to mouse!")
    }
    
    cat(paste0(">>> Genes to keep: ", length(genes_to_keep), " <<<\n"))
    
    # 只保留需要的基因
    seurat_obj <- subset(seurat_obj, features = genes_to_keep)
    rename_vector <- rename_vector[genes_to_keep]
    
    # 使用RenameFeatrues重命名基因
    seurat_obj <- RenameFeatures(seurat_obj, newnames = rename_vector)
    
    cat(paste0(">>> Mapped ", length(genes_to_keep), " genes to mouse orthologs <<<\n"))
    cat(paste0(">>> Unmapped genes: ", length(res$geneID_unmapped), " <<<\n"))
    
    # rat使用mouse数据库
    organism <- "mouse"
  }
  # --- 修改结束 ---
  # 1. 设置数据库
  if (organism == "human") {
    CellChatDB <- CellChatDB.human
  } else {
    CellChatDB <- CellChatDB.mouse
  }
  
  # 2. 创建 CellChat 对象
  cellchat <- createCellChat(seurat_obj, 
                             meta = seurat_obj@meta.data, 
                             group.by = celltype_col)
  cellchat@DB <- CellChatDB
  
  # 3. 预处理流程
  cellchat <- subsetData(cellchat)
  cellchat <- identifyOverExpressedGenes(cellchat)
  cellchat <- identifyOverExpressedInteractions(cellchat)
  
  # 4. 计算通讯概率
  cellchat <- computeCommunProb(cellchat, population.size = TRUE)
  cellchat <- filterCommunication(cellchat) 
  
  # 5. 通路级计算与聚合
  cellchat <- computeCommunProbPathway(cellchat)
  cellchat <- aggregateNet(cellchat)
  
  # 6. 保存结果 (保存为 cco.PD.rds 格式)
  file_name <- paste0("cco.", group_name, ".rds")
  saveRDS(cellchat, file = file_name)
  cat(paste0(">>> Saved to: ", file_name, " <<<\n\n"))
  
  # 7. 返回对象供赋值使用
  return(cellchat)
}

  } else 0
  
  # 基础尺寸
  width  <- n_x * 0.3
  height <- n_y * 0.22
  
  
  
  cat("自动设定尺寸：宽 =", round(width,1), "英寸，高 =", round(height,1), "英寸\n")

# run_pyscenic_csv.sh

INPUT="count_mat.loom"
THREADS=32
SPECIES=$1
log=$2

abs_log=$(realpath -m "$log")
#SPECIES 1 为human 2为mouse
# 记录程序开始时间
start_time=$(date +"%Y-%m-%d %H:%M:%S")
start_timestamp=$(date +%s)

echo "程序开始时间: $start_time"
echo "运行命令: pyscenic"
echo "======================================"


source /BioData/00.Software/12.anaconda/anaconda3/etc/profile.d/conda.sh

conda activate pyscenic

# 设置 PySCENIC 变量
if [ "$SPECIES" == "1" ]; then
    echo "正在为 Human 设置 PySCENIC 路径..."
    TFS="/BioData/08.ref/pysenic/allTFS_hg38.txt"
    DB_DIR="/BioData/08.ref/pyscenic/human/hg38_10kbp_up_10kbp_down_full_tx_v10_clust.genes_vs_motifs.rankings.feather"
    MOTIFS="/BioData/08.ref/pyscenic/motifs-v10nr_clust-nr.hgnc-m0.001-o0.0.tbl"
elif [ "$SPECIES" == "2" ]; then
    echo "正在为 Mouse 设置 PySCENIC 路径..."
    TFS="/BioData/08.ref/pyscenic/allTFS_mm.txt"
    DB_DIR="/BioData/08.ref/pyscenic/mm/mm10_10kbp_up_10kbp_down_full_tx_v10_clust.genes_vs_motifs.rankings.feather"
    MOTIFS="/BioData/08.ref/pyscenic/motifs-v10nr_clust-nr.mgi-m0.001-o0.0.tbl"
fi

# 1. GRN 推断
pyscenic grn \
    --num_workers $THREADS \
    --method grnboost2 \
    -o adjacencies.tsv \
    $INPUT \
    $TFS

# # 2. 调控模块识别
pyscenic ctx \
    adjacencies.tsv \
    $DB_DIR \
    --annotations_fname $MOTIFS \
    --expression_mtx_fname $INPUT \
    --output regulons.csv \
    --num_workers $THREADS

# 3. AUCell 评分
pyscenic aucell \
    $INPUT \
    regulons.csv \
    -o auc_mtx.loom \
    --num_workers $THREADS


# 记录程序结束时间
end_time=$(date +"%Y-%m-%d %H:%M:%S")
end_timestamp=$(date +%s)

# 计算运行时间
total_seconds=$((end_timestamp - start_timestamp))
hours=$((total_seconds / 3600))
minutes=$(((total_seconds % 3600) / 60))
seconds=$((total_seconds % 60))

echo "======================================"
echo "程序结束时间: $end_time"
echo "总运行时间: ${hours}小时 ${minutes}分钟 ${seconds}秒"

conda activate email
python /BioData/09.script/send_email_function.py "$abs_log"
