save_plots <- function(
    filename,      # åŸºç¡€æ–‡ä»¶å(ä¸å«æ‰©å±•å)
    plot,          # è¦ä¿å­˜çš„å›¾å½¢å¯¹è±¡(ggplotæˆ–gridå¯¹è±¡)
    path = ".",    # ä¿å­˜è·¯å¾„,é»˜è®¤å½“å‰ç›®å½•
    width = 7.2,   # å®½åº¦(è‹±å¯¸)
    height = 4.8,  # é«˜åº¦(è‹±å¯¸)
    dpi = 600,     # PNGåˆ†è¾¨ç‡
    formats = c("pdf", "png")  # æ”¯æŒä¿å­˜çš„æ ¼å¼
) {
  library(ggplot2)
  library(grid)
  
  # åˆ›å»ºç›®å½•(å¦‚æœä¸å­˜åœ¨)
  if (!dir.exists(path)) {
    dir.create(path, recursive = TRUE)
  }
  
  # åˆ¤æ–­æ˜¯å¦ä¸ºggplotå¯¹è±¡
  is_ggplot <- inherits(plot, "ggplot")
  
  # éå†æ‰€æœ‰éœ€è¦ä¿å­˜çš„æ ¼å¼
  for (fmt in formats) {
    full_path <- file.path(path, paste0(filename, ".", fmt))
    
    if (is_ggplot) {
      # å¦‚æœæ˜¯ggplotå¯¹è±¡,ä½¿ç”¨ggsave
      switch(fmt,
             "pdf" = ggsave(full_path, plot, width = width, height = height),
             "png" = ggsave(full_path, plot, width = width, height = height, dpi = dpi, bg = "white"),
             "tiff" = ggsave(full_path, plot, width = width, height = height, dpi = dpi, bg = "white"),
             warning(paste("Unsupported format:", fmt))
      )
    } else {
      # å¦‚æœæ˜¯gridå¯¹è±¡(å¦‚pheatmap),ä½¿ç”¨ä¼ ç»Ÿå›¾å½¢è®¾å¤‡
      switch(fmt,
             "pdf" = {
               pdf(full_path, width = width, height = height)
               print(plot)
               dev.off()
             },
             "png" = {
               png(full_path, width = width * dpi, height = height * dpi, res = dpi, bg = "white")
               print(plot)
               dev.off()
             },
             "tiff" = {
               tiff(full_path, width = width * dpi, height = height * dpi, res = dpi, bg = "white")
               print(plot)
               dev.off()
             },
             warning(paste("Unsupported format:", fmt))
      )
    }
    
    message(paste("Saved:", full_path))
  }
  
  invisible(NULL)
}
